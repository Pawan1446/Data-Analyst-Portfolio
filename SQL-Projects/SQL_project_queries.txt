--1. What is the total number of rows in each of the 3 tables in the database?

	select count(customer_id) total_rows from Customer

	select count(cust_id) total_rows  from Transactions

	select count(prod_cat_code) total_rows from prod_cat_info


--2. What is the total number of transactions that have a return?

	select count(transaction_id) trans_return from Transactions
	where Qty < 0


--3. As you would have noticed, the dates provided across the datasets are not in a 
--correct format. As first steps, pls convert the date variables into valid date formats 
--before proceeding ahead.
	
	select *,CONVERT(date,tran_date,103) from Transactions

	select *,CONVERT(date,DOB,103) from Customer


--4. What is the time range of the transaction data available for analysis? Show the 
--output in number of days, months and years simultaneously in different columns.

	select 
	min(tran_date) as [start_date],
	max(tran_date) as last_date,
	DATEDIFF(day,min(tran_date),max(tran_date)) as diff_in_days,
	DATEDIFF(month,min(tran_date),max(tran_date)) as diff_in_month,
	DATEDIFF(year,min(tran_date),max(tran_date)) as diff_in_year
	from Transactions


--5. Which product category does the sub-category “DIY” belong to?

	select prod_cat from prod_cat_info
	where prod_subcat = 'DIY'


/*DATA ANALYSIS*/

--1. Which channel is most frequently used for transactions?

	Select top 1 Store_type,count(transaction_id) channel from Transactions
	group by Store_type
	order by count(transaction_id) desc


--2. What is the count of Male and Female customers in the database?

	select Gender,count(customer_Id) Cust_count from Customer
	where gender is not null
	group by Gender


--3. From which city do we have the maximum number of customers and how many?
	
	select top 1 city_code,COUNT(customer_Id) cust_count from  Customer
	where city_code is not null
	group by city_code
	order by COUNT(customer_Id) desc


--4. How many sub-categories are there under the Books category?

	select prod_cat,COUNT(prod_sub_cat_code) sub_cat from prod_cat_info
	where prod_cat = 'BOOKS'
	group by prod_cat


--5. What is the maximum quantity of products ever ordered? 

	select top 1 prod_cat_code,sum(Qty) total_qty from Transactions
	group by prod_cat_code
	order by sum(Qty) desc


--6. What is the net total revenue generated in categories Electronics and Books? 

	select P.prod_cat,sum(t.total_amt) revenue from Transactions T INNER JOIN prod_cat_info P
	ON P.prod_cat_code=T.prod_cat_code
	WHERE P.prod_cat in ('ELECTRONICS','BOOKS')
	group by p.prod_cat


--7. How many customers have >10 transactions with us, excluding returns?

	select count(cust_id) customers from(
	select cust_id,count(transaction_id) cust_count from Transactions
	where Qty > 0
	group by cust_id
	having count(transaction_id) > 10 ) x
	

--8. What is the combined revenue earned from the “Electronics” & “Clothing” 
--categories, from “Flagship stores”?

	
	select T.Store_type,sum(t.total_amt) revenue from Transactions T INNER JOIN prod_cat_info P
	ON P.prod_cat_code=T.prod_cat_code
	WHERE P.prod_cat in ('ELECTRONICS','BOOKS') and T.Store_type='Flagship store'
	group by t.Store_type

	

--9. What is the total revenue generated from “Male” customers in “Electronics” 
--category? Output should display total revenue by prod sub-cat.

	select P.prod_subcat,sum(t.total_amt) revenue from prod_cat_info P INNER JOIN Transactions T
	ON P.prod_cat_code=T.prod_cat_code
	inner join Customer c on c.customer_Id=t.cust_id
	WHERE P.prod_cat ='ELECTRONICS' and c.Gender='m'
	group by p.prod_subcat

--10.What is percentage of sales and returns by product sub category; display only top 
--5 sub categories in terms of sales?

select A.prod_subcat_code,[%change],[%change_return] from(
	select top 5 prod_subcat_code,sum(total_amt)/(select sum(total_amt) from Transactions)*100 as [%change]
	from Transactions
	where Qty > 0
	group by prod_subcat_code
	order by sum(total_amt) desc) as A
	join
	(select top 5 prod_subcat_code,sum(total_amt)/ (select sum(total_amt) from Transactions)*100 [%change_return] 
	from Transactions
	where Qty < 0
	group by prod_subcat_code
	order by sum(total_amt)
	) as B
	on A.prod_subcat_code = B.prod_subcat_code
	


--11. For all customers aged between 25 to 35 years find what is the net total revenue 
--generated by these consumers in last 30 days of transactions from max transaction 
--date available in the data

	select cust_id,tran_date, DATEDIFF(year,C.DOB,GETDATE())as age, sum(total_amt) net_revenue
	from Customer c left join Transactions t on c.customer_Id=t.cust_id 
	where DATEDIFF(year,C.DOB,GETDATE()) between 25 and 35 and
	t.tran_date > (select (dateadd(DAY,-30,max(tran_date))) from Transactions)
	group by cust_id,DATEDIFF(year,C.DOB,GETDATE()),tran_date



--12.Which product category has seen the max value of returns in the last 3 months of 
--transactions?

	select top 1 prod_cat_code,tran_date,sum(total_amt) return_amt from Transactions
	where Qty < 0 and tran_date > (select (dateadd(MONTH,-3,max(tran_date))) from Transactions)
	group by prod_cat_code,tran_date
	having sum(total_amt) is not null
	order by sum(total_amt) 


--13.Which store-type sells the maximum products; by value of sales amount and by 
--quantity sold? 

	select TOP 1 Store_type,sum(Qty) total_qty,sum(total_amt) sales_amt from Transactions
	group by Store_type,Qty
	ORDER BY SUM(total_amt) DESC,sum(Qty) desc


--14.What are the categories for which average revenue is above the overall average. 

	Select t.prod_cat_code,p.prod_cat, avg(total_amt) cat_avg from Transactions t 
	inner join prod_cat_info p on p.prod_cat_code=t.prod_cat_code
	group by t.prod_cat_code,p.prod_cat
	having avg(total_amt) > (select avg(total_amt) from Transactions)
	order by prod_cat_code 


--15. Find the average and total revenue by each subcategory for the categories which 
--are among top 5 categories in terms of quantity sold.

with top_categories 
as (
select top 5 prod_cat_code,sum(qty) as qty_sold from Transactions
group by prod_cat_code
order by sum(Qty) desc
)
select prod_cat_code,prod_subcat_code,sum(total_amt) total_revenue,avg(total_amt) average
from Transactions
where prod_cat_code in (select prod_cat_code from top_categories)
group by prod_cat_code,prod_subcat_code
order by prod_cat_code,prod_subcat_code
